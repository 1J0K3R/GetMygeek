CREATE TABLE Consultant (
    IdConsultant INT AUTO_INCREMENT PRIMARY KEY,
    Nom VARCHAR(50) NOT NULL,
    tel VARCHAR(10),
    mail VARCHAR(100),
    ddcUrl VARCHAR(400),
    idMission INT,
    ResponsableCarriere VARCHAR(50),
    anciennete INT,
    disponibilite VARCHAR(20),
    idPrefDomain INT,
    ImageUrl VARCHAR(400),
    Trigramme VARCHAR(5),
    FOREIGN KEY (idMission) REFERENCES Mission(idMission),
    FOREIGN KEY (idPrefDomain) REFERENCES PreferenceDomaine(idPrefDomain)
);


CREATE TABLE Mission (
    idMission INT AUTO_INCREMENT PRIMARY KEY,
    entreprise VARCHAR(50),
    responsableMission VARCHAR(50),
    ficheDePoste VARCHAR(400),
);


CREATE TABLE PreferenceDomaine (
    idPrefDomain INT AUTO_INCREMENT PRIMARY KEY,
    dev BOOLEAN,
    fonctionnel BOOLEAN,
    infra BOOLEAN
);
